🐍 Python Script – Car Sales Analysis

import pandas as pd

# Load dataset
df = pd.read_excel("CarSalesByModelStart.xlsx")

# 1. Total quantity sold by Dealer ID
dealer_qty = df.groupby("Dealer ID")["Quantity Sold"].sum().sort_values(ascending=False)

# 2. Total profit by Model (lowest first)
model_profit = df.groupby("Model")["Profit"].sum().sort_values()

# 3. Hudson profit by Dealer ID
hudson_profit = df[df["Model"] == "Hudson"].groupby("Dealer ID")["Profit"].sum().sort_values(ascending=False)

# Display results
print("Dealer Quantity Sold (Top):\n", dealer_qty.head())
print("\nModel Profit (Lowest first):\n", model_profit)
print("\nHudson Profit by Dealer (Highest first):\n", hudson_profit.head())
